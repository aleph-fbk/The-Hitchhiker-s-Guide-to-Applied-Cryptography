\NewDocumentCommand\boxForNotes{m O{#1} O{grisfondo} O{primary} O{number within=section}} {
	\newtcbtheorem[#5]{#1}{#2} {
		enhanced,
		breakable,
		colback = #3,
		frame hidden,
		boxrule = 0sp,
		borderline west = {2pt}{0pt}{#4},
		sharp corners,
		detach title,
		before upper = \tcbtitle\par\smallskip,
		coltitle = #4,
		fonttitle = \bfseries\sffamily,
		separator sign none,
		segmentation style={solid, #4}
	}
	{th}
}
